#!/bin/bash

# Define path to the tsm command
TSM_COMMAND="/path/to/tsm"

# Function to start Tableau
start_tableau() {
    echo "Starting Tableau..."
    $TSM_COMMAND start
}

# Start Tableau
start_tableau
